# Cell Styling Example

<Playground>
  <MaterialTable
    columns={[
      {title: 'Name', field: 'name'},
      {title: 'Surname', field: 'surname'},

            color: 'red'
          }
        }
      }},

    ]}
    title="Cell Styling"
  />
